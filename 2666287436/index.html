Go 语言创建网络工具

发表于 2024-07-15 更新于 2024-07-23 分类于 Go ， 5.网络编程 阅读次数： 本文字数： 501 阅读时长 ≈ 2 分钟

IP 地址类型

在 net 包中的 IP 地址类型为字节数组，可以使用 net.ParseIP 函数来将字符串转为 net.IP 类型，如果不能转换则返回 nil：

package main

import (
	"fmt"
	"net"
)

func main() {
	ip := net.ParseIP("128.0.0.255")
	if ip == nil {
		fmt.Println("Invalid IP address")
		return
	}
	fmt.Printf("%T: %v\n", ip, ip) // 输出：net.IP: 128.0.0.255
	// 支持 IPv6 地址
	fmt.Println(net.ParseIP("1:0:0:0:0:0:0:1")) // 输出：1::1
}

子网掩码计算

由于子网掩码是个特殊的 IP 地址，所以和 IP 地址类型一样是个字节数组。可以通过 net.CIDRMask 函数来生成 net.IPMask 类型子网掩码，并能通过 net.IP 类型的 Mask 方法来计算网络地址：

package main

import (
	"fmt"
	"net"
)

func main() {
	ip := net.ParseIP("192.168.1.1")
	// 生成子网掩码 255.255.0.0
	ipMask := net.CIDRMask(16, 32)
	fmt.Println(ipMask) // 输出：ffff0000
	// 计算网络地址
	fmt.Println(ip.Mask(ipMask)) // 输出：192.168.0.0
}

有了 IP 地址和子网掩码，就可以计算出网段内所有可用 IP 地址：

package main

import (
	"fmt"
	"net"
)

// 计算网络地址
func networkAddress(ip net.IP, mask net.IPMask) net.IP {
	return ip.Mask(mask)
}

// 计算广播地址
func broadcastAddress(ip net.IP, mask net.IPMask) net.IP {
	ip = ip.To4()
	broadcast := make(net.IP, len(ip))
	for i := 0; i < len(ip); i++ {
        // 用 IP 地址的每一段与对应子网掩码段取反后的结果进行按位或操作
		broadcast[i] = ip[i] | ^mask[i]
	}
	return broadcast
}

// 打印出所有可用的 IP 地址
func printAvailableIPs(network net.IP, broadcast net.IP) {
	incrementIP := func(ip net.IP) {
		for j := len(ip) - 1; j >= 0; j-- {
			ip[j]++
			if ip[j] > 0 {
				break
			}
		}
	}

	for ip := net.ParseIP(network.String()); !ip.Equal(broadcast); incrementIP(ip) {
		fmt.Println(ip)
	}
}

func main() {
	ip := net.ParseIP("192.168.1.130")
	mask := net.CIDRMask(30, 32) // 子网掩码为 255.255.255.252

	network := networkAddress(ip, mask)
	broadcast := broadcastAddress(ip, mask)

	fmt.Println("Network Address:", network)
	fmt.Println("Broadcast Address:", broadcast)
	fmt.Println("Available IPs:") // 加上广播地址，一共 4 个可用地址
	printAvailableIPs(network, broadcast)
}

域名解析

要将域名解析为 IP 地址，可以使用 net.ResolveIPAddr 或 net.LookupHost 函数：

package main

import (
	"fmt"
	"net"
)

func main() {
	hostName := "example.com"
	// 如果要解析出 IPv6 地址，第一个参数输入 ip6
	addr, _ := net.ResolveIPAddr("ip", hostName)
	fmt.Println(addr)
    // 如果要解析的域名对应多个 IP 地址，用下面方法返回地址列表
	addrs, _ := net.LookupHost(hostName)
	fmt.Println(addr)
}
