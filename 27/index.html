HTTP 客户端

发表于 2024-06-27 更新于 2024-07-23 分类于 Go ， 5.网络编程 阅读次数： 本文字数： 3.2k 阅读时长 ≈ 11 分钟

发起请求

在 Go 语言中，标准库 net/http 提供了发送 HTTP 请求所需的一切功能。每次发送请求后，需要手动关闭响应体（response body），防止资源泄漏。

发送 GET 请求

可以使用高度封装的 http.Get 函数快速发送一个 GET 请求，只需要传入访问地址：

package main

import (
	"net/http"
)

func main() {
	// 使用 GET 请求访问 example.com
	response, _ := http.Get("https://example.com")
	defer response.Body.Close()
}

发送 POST 请求

同样使用快捷方法来发送 POST 请求，除了指定 URL 地址外，还需要指定 MIME 类型（请求体内容类型）和请求体内容（io.Reader 类型数据）：

package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"strings"
)

func main() {
	// 发送 POST 请求三个参数：地址、数据类型、数据内容
	testData := strings.NewReader("hello")
	response, _ := http.Post("https://httpbin.org/post", "text/plain", testData)
	defer response.Body.Close()
	body, _ := io.ReadAll(response.Body)
	fmt.Printf("%s\n", body)

	// 发送 JSON 数据
	data := map[string]interface{}{
		"name": "John",
		"age":  30,
	}
	jsonData, _ := json.Marshal(data)
	response, _ = http.Post("https://httpbin.org/post", "application/json", bytes.NewBuffer(jsonData))
	defer response.Body.Close()
	body, _ = io.ReadAll(response.Body)
	fmt.Printf("%s\n", body)
}

发送自定义请求

上面用到的 http.Get 和 http.Post 都是基于 http.Request 类型构建，因此可以通过 http.NewRequest 来直接构建一个自定义请求：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	client := &http.Client{}
	// 构建一个 HEAD 请求，必须指定请求方法和地址
	request, _ := http.NewRequest("HEAD", "https://localhost:12345", nil)
	// client 调用 Do 方法发起请求
	response, _ := client.Do(request)
	fmt.Println(response.StatusCode)
}

访问请求字段

http.Request 结构体有许多字段，可以通过这些字段访问和修改请求的各种属性：

package main

import (
	"bytes"
	"fmt"
	"net/http"
)

func main() {
	client := &http.Client{}
	req, _ := http.NewRequest("HEAD", "https://localhost:12345/123", bytes.NewBufferString(`{"name":"John","age":30}`))

	// 获取请求方法
	fmt.Println(req.Method) // 输出：HEAD
	// 修改请求方法
	req.Method = "POST"

	// 获取请求地址和路径
	fmt.Println(req.URL.Host) // 输出：localhost:12345
	fmt.Println(req.URL.Path) // 输出：/123

	// 获取请求体。读取后失效，必须重新设置请求体
	fmt.Println(req.Body) // 输出：{{"name":"John","age":30}}
	// 获取请求表单
	fmt.Println(req.Form) // 输出：map[]

	// 获取请求头
	for k, v := range req.Header {
		fmt.Printf("%s: %s\n", k, v)
	}

	// client 调用 Do 方法发起请求
	response, _ := client.Do(req)
	fmt.Println(response.StatusCode)
}

发送表单数据

表单数据可以使用 url.Values 结构，通过 http.PostForm 方法来发送：

package main

import (
	"fmt"
	"io"
	"net/http"
	"net/url"
)

func main() {
	// 构造表单数据
	formData := url.Values{
		"name": {"John"},
		"age":  {"30"},
	}
	// 也能通过 Set 方法来设置内容
	formData.Set("username", "john_doe")
	// 如果 key 已存在则直接修改内容
	formData.Set("name", "doe")
	response, _ := http.PostForm("https://httpbin.org/post", formData)
	defer response.Body.Close()

	// 通过请求体返回查询发送的完整表单数据
	body, _ := io.ReadAll(response.Body)
	fmt.Printf("%s\n", body)
}

发送多个文件

发送文件需要使用 multipart.Writer 来构建 multipart/form-data 类型的请求体，还允许包含额外表单字段：

package main

import (
	"bytes"
	"fmt"
	"io"
	"mime/multipart"
	"net/http"
	"os"
)

func main() {
	// 创建一个buffer，用于存储构造的请求体
	body := &bytes.Buffer{}
	// 通过 multipart.NewWriter 创建一个 multipart writer 实例，用于构建 multipart/form-data 类型的请求体
	writer := multipart.NewWriter(body)

	// 准备发送的文件列表，假设这两个文件都存在
	files := []string{"verified_contents.json", "new.html"}
	// 循环处理每一个文件
	for _, path := range files {
		file, _ := os.Open(path)
		defer file.Close()
		// 使用 CreateFormFile 方法，为每个文件创建一个表单文件字段。字段名 files 需要根据实际调整
		part, _ := writer.CreateFormFile("files", file.Name())
		// 将文件数据拷贝到表单数据部分
		_, _ = io.Copy(part, file)
	}

	// 添加其他额外的表单字段
	writer.WriteField("username", "exampleUser")
	// 关闭 writer，在请求体的末尾添加结束边界
	writer.Close()

	// 创建 POST 请求
	req, _ := http.NewRequest("POST", "https://httpbin.org/post", body)
	// 设置 Content-Type 为 multipart/form-data 和边界参数
	req.Header.Set("Content-Type", writer.FormDataContentType())

	// 发送请求，读取响应
	client := &http.Client{}
	resp, _ := client.Do(req)
	defer resp.Body.Close()
	response, _ := io.ReadAll(resp.Body)
	fmt.Println(string(response))
}

请求重试

在请求发送失败时，可以简单地使用循环来重试请求，但需要注意设置退出条件，重试间隔时间也要随重试次数指数增长：

package main

import (
	"fmt"
	"net/http"
	"time"
)

func main() {
	url := "https://www.httpbin.org/status/500"
	maxRetries := 5
	var resp *http.Response
	var err error

	// 重试循环次数
	for i := 0; i < maxRetries; i++ {
		resp, err = http.Get(url)
		// 请求正常时退出循环
		if err == nil && resp.StatusCode == http.StatusOK {
			fmt.Println("Request successful")
			defer resp.Body.Close()
			break
		}

		// 时刻记得关闭请求
		if resp != nil {
			resp.Body.Close()
		}

		// 指数退避：等待 2 的 i 次幂秒
		waitTime := time.Duration(1<<i) * time.Second
		fmt.Printf("Retry %d after %v due to error: %v\n", i+1, waitTime, err)
		time.Sleep(waitTime)
	}

	// 有错报错，没错报状态码
    if err != nil {
		fmt.Println("After retries, the final error is:", err)
	} else {
		fmt.Println("Response Status Code:", resp.StatusCode)
	}
}

也可以使用开源库 retry-go 或 backoff 来实现更复杂的重试逻辑：

package main

import (
	"fmt"
	"github.com/avast/retry-go"
	"net/http"
	"time"
)

func makeRequest(url string) (*http.Response, error) {
	resp, err := http.Get(url)
	if err != nil {
		return nil, err
	}
	if resp.StatusCode != http.StatusOK {
		resp.Body.Close() // 确保关闭非200响应的body
		return nil, fmt.Errorf("received non-200 status code: %v", resp.StatusCode)
	}
	return resp, nil
}

func main() {
	url := "https://www.httpbin.org/status/505"
	var resp *http.Response
	err := retry.Do(
		func() error {
			var err error
			resp, err = makeRequest(url)
			return err
		},
		retry.Attempts(3),
		retry.Delay(time.Second),
		retry.MaxDelay(time.Minute),
		retry.OnRetry(func(n uint, err error) {
			fmt.Printf("Retry %d: error: %v\n", n+1, err)
		}),
	)

	if err != nil {
		fmt.Println("Request failed after retries:", err)
		return
	}
	defer resp.Body.Close()
	fmt.Println("Request successful:", resp.StatusCode)
}

读取响应

在使用 net/http 包发送 HTTP 请求后会返回 http.Response 结构体，这个结构体封装响应的所有内容。

响应体

从服务器返回的响应主体数据，类型为 io.ReadCloser，所以可以用标准的 io 库来读取：

package main

import (
	"fmt"
	"io"
	"net/http"
)

func main() {
	// 使用 GET 请求访问 example.com
	response, _ := http.Get("https://example.com")
	defer response.Body.Close()

	// 通过 io.ReadAll 来读取响应体，适用于响应体不是非常大的情况
	c
	// 返回的字节数组
	fmt.Println(body)
	// 转为字符串格式化输出，也能调用 string(body) 来转换
	fmt.Printf("%s\n", body)
}

对于大的响应体，可能需要使用流式读取来防止内存用量过大。

响应状态

响应状态码通过 resp.StatusCode 获得，可以同 http.StatusOK 进行比较得到响应是否成功的结果。另外还有 resp.Status 类型保存状态码和状态文本的组合：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	resp, _ := http.Get("https://example.com/404")
	defer resp.Body.Close()

	// 假设请求的地址不存在
	if resp.StatusCode != http.StatusOK {
		fmt.Println("Error: Status code is not 200:", resp.StatusCode) // 输出：404
		fmt.Println("Response status:", resp.Status)                   // 输出：404 Not Found
	}
}

响应头部

响应头部包含一系列键值对，可以用来获取如内容类型、内容长度、服务器信息等详细数据：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	resp, _ := http.Get("https://example.com/404")
	defer resp.Body.Close()

	// 获取响应头部中所有项
	for k, v := range resp.Header {
		fmt.Printf("%s: %s\n", k, v)
	}

	// 单独获取响应头部中某一项值
	fmt.Println("Content-Type:", resp.Header.Get("Content-Type"))
}

响应长度

响应体的长度由 resp.ContentLength 获取。如果未知则为 -1：

package main

import (
	"fmt"
	"io"
	"net/http"
	"strings"
)

func main() {
	resp, _ := http.Post("https://httpbin.org/post", "text/plain", strings.NewReader("hello"))
	defer resp.Body.Close()

	// 获取响应体的长度
	fmt.Println("Content Length:", resp.ContentLength)
	// 验证一下响应内容
	fmt.Println("Content-Type:", resp.Header.Get("Content-Type"))
	body, _ := io.ReadAll(resp.Body)
	fmt.Printf("%s\n", body)
}

响应 Cookie

Cookie 通常用于服务器识别或追踪用户会话，使用 resp.Cookies 方法可以获取响应相关的 []*http.Cookie 切片，每个元素代表一个 Cookie：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	// 使用谷歌示例，会返回一大串 Cookie
	resp, _ := http.Get("https://google.com")
	defer resp.Body.Close()

	// 获取所有 Cookie
	cookies := resp.Cookies()
	if len(cookies) == 0 {
		fmt.Println("No cookies received")
	} else {
		fmt.Println("Received Cookies:")
		for _, cookie := range cookies {
			fmt.Printf("- Name: %s, Value: %s, Expires: %s, Domain: %s, Path: %s, HttpOnly: %t, Secure: %t\n",
				cookie.Name, cookie.Value, cookie.Expires, cookie.Domain, cookie.Path, cookie.HttpOnly, cookie.Secure)
		}
	}
}

设置客户端

可以使用自定义客户端来发送请求，通过配置 http.Transport 和 http.Client 的其他字段能自定义客户端的行为。

设置超时

为了防止网络请求无限期挂起，可以为 http.Client 设置超时：

package main

import (
	"fmt"
	"net/http"
	"time"
)

func main() {
	client := &http.Client{
		Timeout: time.Millisecond * 1, // 设置超时为 1 毫秒
	}

	request, _ := http.NewRequest("PUT", "https://www.baidu.com", nil)
	res, err := client.Do(request)
	if err != nil {
		fmt.Println(res) // res 此时为 <nil>
		panic(err) // 报错： context deadline exceeded (Client.Timeout exceeded while awaiting headers)
	}
}

Timeout 设定了客户端等待响应的最大时间，这个时间包括连接时间、重定向和读取响应体的时间。超过这个时间，客户端将取消请求，返回错误。

处理重定向

CheckRedirect 是 http.Client 的一个可选参数，其类型是一个函数，用于控制处理 HTTP 重定向的行为。如果服务器返回了重定向响应（如状态码为 301、302、303、307 或 308），http.Client 将调用 CheckRedirect 函数来决定是否遵循这个重定向。例如，限制重定向的次数以避免重定向循环，或者只在重定向到特定域名时才继续：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	client := &http.Client{
		CheckRedirect: func(req *http.Request, via []*http.Request) error {
			// 超过次数停止重定向
			if len(via) >= 3 {
				return http.ErrUseLastResponse
			}
			fmt.Println("Redirecting to:", req.URL.String())
			// 返回 nil 会继续重定向
			return nil 
		},
	}

	// 会发生 5 次跳转的地址
	request, _ := http.NewRequest("GET", "https://www.httpbin.org/absolute-redirect/5", nil)
	res, _ := client.Do(request)
	defer res.Body.Close()
}

设置 Cookie

http.Client 的 Jar 参数传入 http.CookieJar 接口的实现，负责存储和发送 cookies 到 HTTP 请求中，以及从 HTTP 响应中提取 cookies。

通过使用 Jar，http.Client 能够像浏览器一样处理 cookies，从而简化了需要管理多次请求和响应中状态持久化的复杂任务。例如自动登录、抓取需要会话支持的网站或自动化测试 web 应用：

package main

import (
	"golang.org/x/net/publicsuffix"
	"net/http"
	"net/http/cookiejar"
)

func main() {
	// 创建一个默认 cookie jar
	jar, _ := cookiejar.New(&cookiejar.Options{PublicSuffixList: publicsuffix.List})
	// 创建 http.Client 并设置 jar
	client := &http.Client{Jar: jar}
	// 发送请求
	resp, _ = client.Get("http://example.com")
	defer resp.Body.Close()
}

实际上，可以直接通过请求头传入原始 cookie 字符串，而不一定要使用 CookieJar：

package main

import (
	"net/http"
)

func main() {
	// 创建一个 HTTP 请求
	req, _ := http.NewRequest("GET", "http://example.com", nil)
	// 设置 Cookie 头
	req.Header.Set("Cookie", "name=value; token=abc123")
	// 创建 http.Client 实例
	client := &http.Client{}
	// 发送请求
	resp, _ := client.Do(req)
	defer resp.Body.Close()
}

另外也能使用 http.Cookie 结构创建一个或多个 Cookie，并使用 AddCookie 方法将 Cookie 添加到请求中：

package main

import (
	"fmt"
	"io"
	"net/http"
)

func main() {
	client := &http.Client{}
	req, _ := http.NewRequest("GET", "https://www.httpbin.org/cookies", nil)

	// 创建多个 Cookie
	cookies := []http.Cookie{
		{Name: "ISSW", Value: "1"},
		{Name: "HSID", Value: "AlQPgfM7zmZSVFLG1"},
		{Name: "SSID", Value: "ArlkB-PwOGhOiyJ_o"},
	}

	// 遍历并将每个 Cookie 添加到请求中。也可以使用 http.SetCookie 来设置
	for _, cookie := range cookies {
		req.AddCookie(&cookie)
	}

	// 发送请求
	resp, _ := client.Do(req)
	defer resp.Body.Close()
	body, _ := io.ReadAll(resp.Body)
	fmt.Printf("%s", body)
}

设置请求头

请求头类型 http.Header 本质上是一个 map[string][]string，键是头字段名，值是字符串切片，表示头字段的多个值：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	// 自定义客户端
	client := &http.Client{}
	req, _ := http.NewRequest("GET", "http://example.com", nil)

	// 要获取头字段的值。如果字段不存在返回空字符串
	_ = req.Header.Get("Content-Type")
	// 加入请求头字段，由于允许同一个头字段出现多次，不会覆盖原值
	req.Header.Add("Accept", "application/json")
	// 修改请求头字段，如果键不存在则新增
	req.Header.Set("Content-Type", "application/json")
	// 删除请求头字段
	req.Header.Del("User-Agent")

	// 定义一个映射，直接赋值给请求头
	headers := map[string][]string{
		"Content-Type":    {"application/json"}, // 即使只有一个值也要用切片
		"X-Custom-Header": {"Value1", "Value2"}, // 支持多值
		"Authorization":   {"Bearer token_here"}, // 支持空格
	}
	req.Header = headers

	// 输出现有请求头来验证
	for k, v := range req.Header {
		fmt.Printf("%s: %+#v\n", k, v)
	}

	// 发送请求
	resp, _ := client.Do(req)
	defer resp.Body.Close()
}

HTTP 协议允许同一个头字段出现多次，可多次使用 Add 来添加，但对一个已存在的头字段使用修改或删除操作时，会影响该头字段的所有值，因此也能通过索引来细粒度控制要操作哪个值。

设置连接

http.Client 的 Transport 参数能设置 HTTP 连接的许多方面，如连接池、代理、TLS 配置等。下面是一些常用配置：

  • MaxIdleConns：指定客户端可以保持的最大空闲连接数。
  • MaxIdleConnsPerHost：为每个主机设置可以保持的最大空闲连接数。
  • IdleConnTimeout：空闲连接关闭前保持时间。
  • DisableCompression：设置为 true 则关闭请求压缩功能。
  • DisableKeepAlives：设置为 true 则禁用持久连接（Keep-Alive）。
  • TLSClientConfig：配置 TLS，可以设置证书，禁用某些加密方法，或者完全关闭证书验证。
  • Proxy：配置代理。
  • ResponseHeaderTimeout：设置个等待服务器响应头的超时时间。

下面是一个简单示例：

package main

import (
	"crypto/tls"
	"net/http"
	"time"
)

func main() {
	// 配置多个字段
	transport := &http.Transport{
		MaxIdleConns:          100,
		MaxIdleConnsPerHost:   10,
		IdleConnTimeout:       90 * time.Second,
		DisableCompression:    true,
		TLSClientConfig:       &tls.Config{InsecureSkipVerify: true}, // 禁用 TLS 验证
		Proxy:                 http.ProxyFromEnvironment,
		ResponseHeaderTimeout: 10 * time.Second,
	}

	// 配置传给客户端
	client := &http.Client{Transport: transport}
	req, _ := http.NewRequest("GET", "http://example.com", nil)

	// 发送请求
	resp, _ := client.Do(req)
	defer resp.Body.Close()
}

调试请求

在 net/http/httputil 中提供的方法 DumpRequestOut 和 DumpResponse 能对请求和响应详情进行查看：

package main

import (
	"fmt"
	"net/http"
	"net/http/httputil"
)

func main() {
	// 自定义客户端
	client := &http.Client{}
	request, err := http.NewRequest("GET", "http://example.com", nil)

	// 获取请求信息
	debugRequest, err := httputil.DumpRequestOut(request, true)
	if err != nil {
		panic(err)
	}
	fmt.Printf("%s\n", debugRequest)

	// 实际发送请求
	response, err := client.Do(request)
	defer response.Body.Close()

	// 获取响应信息
	debugResponse, err := httputil.DumpResponse(response, true)
	if err != nil {
		panic(err)
	}
	fmt.Printf("%s\n", debugResponse)
}
