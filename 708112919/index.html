Go 语言中配置管理

发表于 2024-06-28 更新于 2024-07-23 分类于 Go ， 4.基本应用 阅读次数： 本文字数： 1k 阅读时长 ≈ 3 分钟

配置文件

在 Go 语言中管理程序配置文件通常使用 JSON、YAML、或者环境变量等格式来存储配置信息。将配置储存在文件中的优点是，可对其进行版本控制，并集成到自动化构建过程中。

JSON 格式

JSON 格式在 Go 语言中对应着带标签的结构体，使用 encoding/json 包中的函数进行转换：

package main

import (
	"encoding/json"
	"fmt"
	"os"
)

type Config struct {
	Host string `json:"host"`
	Port int    `json:"port"`
}

// WriteConfig 方法将配置写入到指定文件
func (config *Config) WriteConfig(path string) error {
	// 使用 MarshalIndent 而不是 Marshal 让最终写入格式好看一点
	data, err := json.MarshalIndent(config, "", "  ")
	if err != nil {
		return err
	}

	// 创建并打开文件
	file, err := os.Create(path)
	if err != nil {
		return err
	}
	defer file.Close()

	// 写入数据后，直接返回 error，如果没有错误则是 nil
	_, err = file.Write(data)
	return err
}

// ReadConfig 方法用于从文件读取配置
func (config *Config) ReadConfig(path string) error {
	// 打开文件
	file, err := os.Open(path)
	if err != nil {
		fmt.Println("Error opening config file:", err)
		return err
	}
	defer file.Close()

	// 解码 JSON 格式到结构体
	err = json.NewDecoder(file).Decode(&config)
	if err != nil {
		fmt.Println("Error decoding config:", err)
		return err
	}
	return err
}

func main() {
	// 配置文件路径
	path := "config.json"
	// 要写入的配置
	configW := Config{"127.0.0.1", 1080}
	// 读取配置后存放的变量
	configR := Config{}

	// 调用写入配置方法
	err := configW.WriteConfig(path)
	if err != nil {
		fmt.Println("Error writing config:", err)
		return
	}

	// 调用读取配置方法，并打印出来
	err = configR.ReadConfig(path)
	if err != nil {
		fmt.Println("Error reading config:", err)
		return
	}
	fmt.Printf("%+#v\n", configR)
}

YAML 格式

YAML 是一种直观的数据序列化格式，支持复杂的数据结构。可以通过第三方库 gopkg.in/yaml.v3 来解析：

package main

import (
	"fmt"
	"gopkg.in/yaml.v3"
	"os"
)

type Config struct {
	Host string `yaml:"host"`
	Port int    `yaml:"port"`
}

// readConfig 函数用于读取配置，没用方法
func readConfig(filename string) (*Config, error) {
	file, err := os.Open(filename)
	if err != nil {
		return nil, err
	}
	defer file.Close()

	// 在函数内部初始化配置
	config := &Config{}
	if err := yaml.NewDecoder(file).Decode(config); err != nil {
		return nil, err
	}

	return config, nil
}

// writeConfig 函数写入配置到文件
func writeConfig(filename string, config *Config) error {
	file, err := os.Create(filename)
	if err != nil {
		return err
	}
	defer file.Close()

	// 编码并写入
	if err := yaml.NewEncoder(file).Encode(config); err != nil {
		return err
	}

	return nil
}

func main() {
	// 写入配置文件
	configWrite := Config{
		Host: "localhost",
		Port: 8080,
	}
	err := writeConfig("config.yaml", &configWrite)
	if err != nil {
		fmt.Println("Error writing YAML:", err)
		return
	}

	// 读取配置文件
	configRead, err := readConfig("config.yaml")
	if err != nil {
		fmt.Println("Error reading YAML:", err)
		return
	}

	// 分别读取配置段落
	fmt.Println("Host:", configRead.Host)
	fmt.Println("Port:", configRead.Port)
}

TOML 格式

TOML（Tom’s Obvious, Minimal Language）格式相比于 JSON 表达能力更强，并且在 Go 语言中转换更容易。采用 TOML 作为配置文件，需要使用 BurntSushi/toml 库来处理 TOML 数据：

package main

import (
	"fmt"
	"github.com/BurntSushi/toml"
	"os"
)

// Config 结构体表示 TOML 配置文件中的数据结构
type Config struct {
	Server   ServerConfig   `toml:"server"`
	Database DatabaseConfig `toml:"database"`
}

// ServerConfig 是服务器配置部分
type ServerConfig struct {
	Host string `toml:"host"`
	Port int    `toml:"port"`
}

// DatabaseConfig 是数据库配置部分
type DatabaseConfig struct {
	User     string `toml:"user"`
	Password string `toml:"password"`
	Database string `toml:"database"`
}

// readConfig 从指定的文件名读取TOML配置
func readConfig(filename string) (*Config, error) {
	var config Config
	// 参数指定配置文件路径和配置结构体变量指针
	_, err := toml.DecodeFile(filename, &config)
	if err != nil {
		return nil, err
	}
	return &config, nil
}

// writeConfig 将配置写入指定的文件名
func writeConfig(filename string, config *Config) error {
	file, err := os.Create(filename)
	if err != nil {
		return err
	}
	defer file.Close()

	// 这次不合并了，NewEncoder 返回一个 Encoder 结构体指针
	encoder := toml.NewEncoder(file)
	// 将配置内容写入到文件
	if err := encoder.Encode(config); err != nil {
		return err
	}
	return nil
}

func main() {
	// 直接初始化一个嵌套结构体
	config := Config{
		Server: ServerConfig{
			Host: "localhost",
			Port: 8080,
		},
		Database: DatabaseConfig{
			User:     "admin",
			Password: "secret",
			Database: "example_db",
		},
	}

	// 写入配置
	err := writeConfig("config.toml", &config)
	if err != nil {
		fmt.Println("Error writing config:", err)
		return
	}

	// 读取配置
	readConfig, err := readConfig("config.toml")
	if err != nil {
		fmt.Println("Error reading config:", err)
		return
	}

	// 通过点操作符获取数据
	fmt.Printf("Server Host: %s\n", readConfig.Server.Host)
	fmt.Printf("Database Name: %s\n", readConfig.Database.Database)
}

读取环境变量

使用环境变量是一种将配置信息从应用程序代码中分离出来的有效方式，在用容器化运行时可以方便地传入参数：

package main

import (
	"fmt"
	"os"
)

func main() {
	host := os.Getenv("Host")
	port := os.Getenv("NUMBER_OF_PROCESSORS") // 用系统变量伪装成端口号
	fmt.Println("Host:", host, "Port:", port)
}
